# CAPTION("Pour une tonalité majeure, et pour la tonalité de DO Majeur du prélude… ")

# --- Gamme de DO ---

SET_CURSOR(220)
gammedo = SCALE('c', {staff:2, speed:4, offset:40})

STAFF(2).write("DO Majeur", {left:266, offset_y:-24})

# CAPTION("ces notes “pivots“ sont le degré 4 — donc la 4e note de la gamme de la tonalité —, le degré 5 — donc la 5e note — ou le degré 7 de la tonalité — donc la sensible de cette tonalité. C'est toujours la 4e, la 5e ou la 7e note qui est altérée pour passer ou tout au moins pour produire le ton voisin.")
WAIT(2)
gammedo.note(4).surround()
WAIT(1)
gammedo.note(5).surround()
WAIT(1)
gammedo.note(7).surround()

WAIT_CAPTION

# CAPTION("Les notes pivots sont donc la 4e, la 5e et la 7e")


WAIT_CAPTION

# CAPTION("Si je monte le 5e degré, j'obtiens le ton voisin du relatif.", {wait:true})

gammedo.note(5).arrow('vers lam', {angle:-90, width:60})

CAPTION("Si je réhausse le quatrième degré, le FA ici, j'obtiens la tonalité de la dominante. Ici, SOL Majeur")

gammedo.note(4).arrow('vers sol', {angle:90, width:60})

# -- Gamme de SOL Majeur --

SET_CURSOR(100)

WAIT_CAPTION

CAPTION("Et enfin, si je baisse le 7e degré, la sensible, j'obtiens le dernier ton voisin, la tonalité de la sous-dominante, donc FA Majeur pour DO Majeur.")
