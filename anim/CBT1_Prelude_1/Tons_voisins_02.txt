# === Analyse du Prélude 1 CBT1 ===
# 
# Parenthèse sur l'harmonie
# Dans cette partie on aborde la question des tonalités (survolé) et
# des tons voisins pour se préparer au travail d'analyse sur le prélude
#
# Cette deuxième partie récapitule les ton voisins en montrant les
# accords.
#

DEFAULT('decompte',0)
DEFAULT('staff_top')
DEFAULT('caption_timer', true)
DEFAULT('next', 200)
#SPEED(20)

#SUITE('CBT1_Prelude1_01-11')

NEW_STAFF(SOL)
NEW_STAFF(SOL)
NEW_STAFF(SOL)

# Définition des pas utilisés pour les accords
# 1
NEXT(100)
# 2
NEXT()
# 3
NEXT()

CAPTION("Résumons ce que nous venons de voir sur les tons proches. En se rappelant que c'est une gymnastique à développer jusqu'à ce qu'il n'y ait presque plus à réfléchir.", {wait:true})

CAPTION("Pour une tonalité de DO Majeur…")

SET_CURSOR(2)
accdo = CHORD('c4 e4 g4 c5', {staff:2})

WAIT_CAPTION

# --- SOL ---

CAPTION("J'obtiens le premier ton voisin, le ton de la dominante, SOL Majeur, en ajoutant un dièse, le Fa dièse.")

accdo.arrow('vers sol',{color:blue, width:80, offset_x:20, offset_y:-4})
SET_CURSOR(3)
accsol = CHORD('g4 b4 d5 g5', {staff:2})
accsol.write("Ton de la Dominante (V)", {offset_x:-60})

WAIT_CAPTION

# --- FA ---

CAPTION("J'obtiens le ton voisin de la sous-dominante en ajoutant un bémol.")

accdo.arrow('vers fa',{color:blue, angle:180, width:80, offset_x:-20, offset_y:-4})
SET_CURSOR(1)
accfa = CHORD('f4 a4 c5 f5', {staff:2})
accfa.write("Ton de la Sous-Dominante (IV)", {offset_x:-80})

WAIT_CAPTION

# --- LA mineur ---

CAPTION("J'obtiens le ton voisin du relatif mineur en ré-haussant le 5e degré.")


accdo.arrow('vers lam',{color:blue, angle:-90, width:40})
SET_CURSOR(2)
acclam = CHORD('a4 c5 e5 a5', {staff:1})
acclam.write("Ton du relatif", {offset_x:-30})

WAIT_CAPTION

# --- Ré mineur ---

CAPTION("On peut avoir en tête aussi la tonalité du deuxième degré — ré mineur ici — qui est une modulation assez fréquente, puisque c'est un ton semi-voisin :")

accdo.arrow('vers rem',{color:blue, angle:90, width:40})
accrem = CHORD('d4 f4 a4', {staff:3})
accrem.write("Ton du second degré (II)", {offset_y:-80, offset_x:-60})

WAIT_CAPTION

CAPTION("… : en altérant une seule note, le do ici, on obtient la gamme harmonique ascendante de Ré mineur")

NEW_STAFF(SOL)
SET_CURSOR(1)
NEXT(70)
MOTIF('d e f g a b cd5 d', {staff:4, offset_x:30})

WAIT_CAPTION
