# === QU\'EST-CE QU\'UN TON VOISIN ? ===

CAPTION(\"Qu\'est-ce qu\'on appelle un “ton voisin” ?… C\'est extrêmement simple : deux tons voisins, ce sont deux tonalités, donc un ensemble de notes, qui ne différent que par une seule note, qui n\'ont qu\'une seule note de différence.\", {wait:true})

CAPTION(\"Cette note qui diffère répond au cycle des quintes ou au cycle des bémols. Mais ce n\'est pas par ce biais qu\'on va l\'aborder, je le fais dans le screencast sur ces cycles.\", {wait:true})

CAPTION(\"Ici, nous allons l\'aborder en disant que pour une tonalité majeure, les notes qui changent sont toujours le degré 4, 5 ou 7 de la tonalité. C\'est toujours la 4e, la 5e ou la 7e note qui est altérée pour produire le ton voisin.\", {wait:true})

CAPTION(\"Par exemple pour DO Majeur…\")
SET_CURSOR(200)
gammedo = SCALE(\'c\', {staff:2, speed:4})

CAPTION(\"Si vous avez suivi les autres screencasts, vous savez que hormis la fondamentale — c\'est-à-dire la première note de la tonalité — les autres degrés les plus importants sont le 4e — le FA en DO Majeur — et le 5e — le SOL en DO majeur\")

gammedo.note([4,5]).surround()

WAIT(10)

CAPTION(\"JE SUIS LÀ\", true)
# WAIT_END_CAPTION (il faudrait implémenter cette commande pour attendre la fin du caption)

CAPTION(\"Si je monte le 5e degré, j\'obtiens le ton voisin du relatif.\")

SET_CURSOR(200)
SCALE(\'a\', {type:\'min_h\', staff:1})

CAPTION(\"Si je réhausse le quatrième degré, le FA ici, j\'obtiens la tonalité de la dominante.\")

SET_CURSOR(200)
SCALE(\'g\')


# RETOUR AU TON VOISIN

CAPTION(\"Un ton voisin, c\'est donc, comme je l\'ai dit, un ton, une tonalité, qui n\'a qu\'une note de différence avec la tonalité de référence… C\'est donc une tonalité qui aura une altération en plus, ou une altération en moins, par rapport à cette tonalité\", {wait:true})


CAPTION(\"Cette pièce de Bach ne présente aucune altération à la clé, on est donc en DO Majeur. Ses tons voisins vont être les tons qu\'on obtient en ajoutant ou en retirant une altération.\", {wait:true})

CLEAR()

SET_CURSOR(200)

SCALE(\'c\', {staff:1})
STAFF(1).write(\"DO Majeur\", {left:130, offset_y:-24})

CAPTION(\"En ajoutant un dièse dans l\'ordre des dièses, on obtient la tonalité de Sol Majeur\")

SET_CURSOR(200)

SCALE(\'g\', {staff:2})
STAFF(2).write(\"SOL Majeur\", {left:130, offset_y:-24})

WAIT(1)

CAPTION(\"Comme on ne peut pas retirer de dièses à la tonalité de DO Majeur — elle n\'en a pas — on ajoute un bémol pour obtenir un autre ton voisin. Et on obtient la tonalité de FA Majeur.\", {wait:true})

SET_CURSOR(200)

SCALE(\'f\', {staff:3})
STAFF(3).write(\"FA Majeur\", {left:130, offset_y:-24})


CAPTION(\"Et le dernier ton proche, c\'est celui du relatif, ici le relatif <u>mineur</u> puisque la tonalité est majeure. C\'est la tonalité de La mineur\", {wait:true})

SET_CURSOR(200)

SCALE(\'a\', {type:\'min_h\', staff:4})
STAFF(4).write(\"LA mineur\", {left:130, offset_y:-24})
